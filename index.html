// Extrait de code de votre projet Web (app.js / scripts)
document.addEventListener("DOMContentLoaded", () => {
    console.log("Portfolio et composants initialisés.");

    // Interaction au survol des cartes
    const projectCards = document.querySelectorAll(".project-card");
    projectCards.forEach(card => {
        card.addEventListener("mousemove", (e) => {
            const rect = card.getBoundingClientRect();
            const x = e.clientX - rect.left - (rect.width / 2);
            const y = e.clientY - rect.top - (rect.height / 2);
            card.style.transform = `translate(${x * 0.002}px, ${y * 0.002}px)`;
        });

        card.addEventListener("mouseleave", () => {
            card.style.transform = "none";
            card.style.transition = "transform 0.4s cubic-bezier(0.25, 1, 0.5, 1)";
        });
    });
});
